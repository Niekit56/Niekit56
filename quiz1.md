import random
import time

fakts = {
    "Какое животное считается символом Австралии?": "Кенгуру",
    "Какое языковое семейство преобладает в Китае?": "Сино-тибетское",
    "Какой газ является наиболее распространенным в атмосфере Земли?": "Азот",
    "Какая столица Швейцарии?": "Берн",
    "Какой океан омывает восточное побережье Австралии?": "Тихий океан",
    "Какая планета считается самой близкой к Солнцу?": "Меркурий",
    "Какое животное является национальным символом России?": "Медведь",
    "Какая страна является самой большой по площади в мире?": "Россия",
    "Кто написал роман 'Преступление и наказание'?": "Фёдор Достоевский",
    "Какое вещество придает растениям зеленый цвет?": "Хлорофилл"
}

right = 0
wrong = 0

start_time = time.time()

a = input("Пройдем викторину?(Да/Нет)")
if a.lower() == "да":
    print("let's go")
else:
    print("so bad(")
    exit()

while fakts:
    random_key = random.choice(list(fakts.keys()))
    print(random_key)
    user_answer = input("Введите ваш ответ: ")
    correct_answer = fakts[random_key]

    if user_answer.lower() == correct_answer.lower():
        print("Правильно!")
        right += 1
    else:
        print("Неправильно.")
        wrong += 1

    del fakts[random_key]

end_time = time.time()
execution_time = end_time - start_time

print("Викторина окончена. Все вопросы отвечены!")
print(f"Вы ответили правильно на {right} и не правильно на {wrong} вопросов.")
print(f"Затраченное время {execution_time} секунд")
